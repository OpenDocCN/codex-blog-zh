def Costfunction(X, y, theta):
    m=len(y)
    
    h_theta = sigmoid(X@theta)
    y_pos = -y.T @ np.log(h_theta)
    y_neg = (1-y).T @ np.log(1-h_theta)
    error = y_pos - y_neg
    
    cost = 1/m * sum(error)
    grad = 1/m * (X.T@(h_theta - y))
    
    return cost[0] , grad
